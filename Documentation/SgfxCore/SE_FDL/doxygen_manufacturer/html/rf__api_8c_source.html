91  case SFX_RF_MODE_TX :
92  // set the RF IC in TX mode : this could be DBPSK100 or 600: this distinction will be done during the RF_API_send
93  // you can add some code to switch on TCXO or warm up quartz for stabilization : in case of frequency drift issue
94  break;
95  case SFX_RF_MODE_RX :
96  // set the RF IC in RX mode : GFSK 600bps is the only mode now with BT=1.0 and delta_f += 800Hz
97  // Enable interrupts on RX fifo
98  // RF IC must configure the SYNCHRO BIT = 0xAAAA and synchro frame = 0xB227
99  break;
100  case SFX_RF_MODE_CS200K_RX :
101  // configure the RF IC into sensing 200KHz bandwidth to be able to read out RSSI level

103  break;

105  // configure the RF IC into sensing 300KHz bandwidth to be able to read out RSSI level
106  // This is poosible to make this carrier sense in 2 * 200Kz if you respect the regulation time for listening
107  // RSSI level will outputed during the RF_API_wait_for_clear_channel function.
108  break;
109 
110  default :
111  break;
112  }
